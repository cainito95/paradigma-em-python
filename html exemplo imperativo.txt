<!-- JavaScript é um exemplo IMPERATIVO pq é onde se controla passo a passo da execução, "o que" percorre por "onde" e "como"-->

<script>
function calcular() {

<!-- pega os elementos -->
  let qtd = document.querySelectorAll(".qtd");
  let preco = document.querySelectorAll(".preco");

<!-- cria uma variavel -->
  let total = 0;

<!-- percorre manualmente -->
  for (let i = 0; i < qtd.length; i++) {

<!-- atualiza o valor -->
    total = total + (qtd[i].textContent * preco[i].textContent);
  }

<!-- altera a página -->
  document.getElementById("totalGeral").textContent = "R$ " + total.toFixed(2);
}
</script>
